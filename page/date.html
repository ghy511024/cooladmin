旧版例子

        <script type="text/javascript">
            $(document).ready(function () {
                $("#date2").datetimepicker();//同时选定日期和时间
            })
        </script>

限制某一段时间不可选


        <script type="text/javascript">
            $(document).ready(function () {
                $("#date2").datetimepicker({//月份中数值需要加1（0代表1月）
                    minDate: new Date(2015, 7, 12),
                    maxDate: new Date(1440518400000)
                });
            })
        </script>

设置默认时间


        <script type="text/javascript">
             $("#date3").datetimepicker({
                    defaultDate: "2015-12-13"
                });
        </script>
