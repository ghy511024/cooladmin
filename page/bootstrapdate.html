默认使用


        <script type="text/javascript">
            $(document).ready(function () {
               $("#date1").datetimepicker();
            })
        </script>

参数设置


        <script type="text/javascript">
            $(document).ready(function () {
                  $("#date2").datetimepicker({
                    showMeridian: true, //分屏
                    autoclose: true, //选完关闭
                    todayBtn: true, //现在时间
                });
            })
        </script>

参数设置2


        <script type="text/javascript">
             $("#date3").datetimepicker({
                    startDate: "2015-12-13", //开始时间
                    endDate: "2015-12-28", //开始时间
                    minuteStep: 3//分钟间隔
                });
        </script>
